-- A ConTeXt document

\usemodule[t-literate-modules]

\setupinteraction[
  state=start,
  color=green,
  stype=bold
]
\placebookmarks[chapter, section, subsection][chapter, section]
\setupinteractionscreen[option=bookmark]

\setuppapersize[A4]

\setupindenting[medium,yes]

% do not indent after itemize
\setupitemgroup[itemize][indentnext=no]

% indent the paragraphs after all sectioning heads 
\setupheads[indentnext=yes] 

% indent paragraphs after formulas if there is a blank space in between
\setupformulas[indentnext=auto] 

\usebtxdataset[literateModulesBib.lua]
\usebtxdefinitions[apa]
\setupbtxlist[apa][
  alternative=paragraph,
  width=fit,
  distance=.5em,
  margin=3em % hanging list
]

\startproduct

\startdocument[title=Literate programming of \ConTeXt\ modules]

\component overview

\component preamble

\component mkivCode

\component mkivTests

\component luaCode

\component luaTests

\component luaTemplates

\component luaTemplateTests

\component lakefiles

\component conclusion

\startbackmatter
  \startchapter[title=Bibliography]
    \placelistofpublications
  \stopchapter
\stopbackmatter

\stopdocument
